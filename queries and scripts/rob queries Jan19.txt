select employees.EmployeeID, employees.EmployeeNo, projects.Name from
ProjectTasks join EmployeeTimesheetItems on ProjectTasks.TaskID = EmployeeTimesheetItems.TaskID
join Projects on ProjectTasks.ProjectID = projects.ProjectID join EmployeeTimesheets on EmployeeTimesheetItems.TimesheetID = EmployeeTimesheets.TimesheetID join Employees on EmployeeTimesheets.EmployeeID = Employees.EmployeeID
where Employees.Email is null and employees.EmployeeNo in ('8858', '8865', '8868', '8904', '8914', '8926', 'AZ2000', 'AZ3812', 'AZ3844', 'AZ3865', 'AZ3937')
group by employees.EmployeeID, employees.EmployeeNo, projects.Name

ProjectTasks 
join EmployeeTimesheetItems on ProjectTasks.TaskID = EmployeeTimesheetItems.TaskID
join ProjectGroups on ProjectGroups.groupid = ProjectTasks.groupid
join ProjectParts on ProjectParts.partid = ProjectGroups.partid
join Projects on ProjectTasks.ProjectID = projects.ProjectID 
join EmployeeTimesheets on EmployeeTimesheetItems.TimesheetID = EmployeeTimesheets.TimesheetID 
join TimesheetPeriods on TimesheetPeriods.TimesheetPeriodID = EmployeeTimesheets.TimesheetPeriodID
join Employees on EmployeeTimesheets.EmployeeID = Employees.EmployeeID

SELECT TOP (30000) [EmployeeID]
      ,[EmployeeNo]
      ,[IsActive]
      ,[ManagerID]
      ,[CompanyID]
      ,[OfficeID]
      ,[AllowOT]
      ,[LastModifiedBy]
      ,[LastModifiedDate]
      ,[Email]
      ,[EmailConfirmed]
      ,[PasswordHash]
      ,[SecurityStamp]
      ,[PhoneNumber]
      ,[PhoneNumberConfirmed]
      ,[TwoFactorEnabled]
      ,[LockoutEndDateUtc]
      ,[LockoutEnabled]
      ,[AccessFailedCount]
      ,[UserName]
      ,[Names]
      ,[LockoutDateTimeUtc]
  FROM [eTimeTrack].[dbo].[Employees]
  where [EmployeeID] not in
  (select distinct [EmployeeID] from [EmployeeTimesheets])
